\section{Objetivos}
\subsection{Objetivos}
\begin{frame}{Objetivos}
\textcolor{blue}{Objetivo Geral}: Controle de TV via \textit{smartphone} 
através da voz
\begin{itemize}
	\item BeagleBone Black
	\begin{itemize}
		\item Servidor de reconhecimento de fala \smallskip
		\item Servidor de síntese de fala \smallskip
		\item Servidor LAMP de armazenamento de comandos \smallskip
	\end{itemize}
	\item Arduino UNO
	\begin{itemize}
		\item Transmissor de Sinais Infravermelhos (IR LED) \smallskip
		\item Receptor de Sinais Infravermelhos (IR Receiver) \smallskip
	\end{itemize}
	\item Android
	\begin{itemize}
		\item Microfone (cliente)
	\end{itemize}
\end{itemize}
\end{frame}
