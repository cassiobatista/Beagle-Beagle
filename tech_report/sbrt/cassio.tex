\section{Equações e teoremas}

\begin{theorem}[Nome do Teorema]
Considere o sistema
\begin{equation}
\begin{array}{rrr}
\dot x&=&A.x+B.u\\[2mm] y&=& C.x+D.u
\end{array}
\end{equation}
\begin{equation}
\left[\begin{array}{c|c}    A & b_1 \\ \hline    c & d_1
\end{array}\right]\quad{\hbox{e}}\quad\left[\begin{array}{c|c}
A & b_2 \\ \hline    c & d_2  \end{array}\right].
\end{equation}
Se $A$ for estável, então o par $\{A,B\}$ é estabilizável e isso
se mantém para qualquer $B$.
\end{theorem}
\begin{proof}
A demonstração deste teorema é trivial e é deixada para os
leitores interessados.
\end{proof}
